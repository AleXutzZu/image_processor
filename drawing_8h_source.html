21 template<typename T>
22 void drawCircle(Image<T> &image, const Point &center, unsigned int radius, T color) {
23 int cx = static_cast<int>(center.getX());
24 int cy = static_cast<int>(center.getY());
25
26 int x = radius;
27 int y = 0;
28 int decision = 1 - radius;
29
30 auto plot = [&](int px, int py) {
31 if (px >= 0 && py >= 0 && py < image.getHeight() && px < image.getWidth()) {
32 image.at(px, py) = color;
33 }
34 };
35
36 while (y <= x) {
37 plot(cx + x, cy + y);
38 plot(cx - x, cy + y);
39 plot(cx + x, cy - y);
40 plot(cx - x, cy - y);
41 plot(cx + y, cy + x);
42 plot(cx - y, cy + x);
43 plot(cx + y, cy - x);
44 plot(cx - y, cy - x);
45
46 y++;
47 if (decision <= 0) {
48 decision += 2 * y + 1;
49 } else {
50 x--;
51 decision += 2 * (y - x) + 1;
52 }
53 }
54 }

65 template<typename T>
66 void drawLine(Image<T> &image, const Point &p1, const Point &p2, T color) {
67 int x1 = static_cast<int>(p1.getX());
68 int y1 = static_cast<int>(p1.getY());
69 int x2 = static_cast<int>(p2.getX());
70 int y2 = static_cast<int>(p2.getY());
71
72 int delta_x = abs(x2 - x1);
73 int delta_y = abs(y2 - y1);
74 int step_x = (x1 < x2) ? 1 : -1;
75 int step_y = (y1 < y2) ? 1 : -1;
76 int error = delta_x - delta_y;
77
78 while (true) {
79 if (x1 >= 0 && y1 >= 0 && y1 < image.getHeight() && x1 < image.getWidth()) {
80 image.at(x1, y1) = color;
81 }
82
83 if (x1 == x2 && y1 == y2) break;
84 int double_error = 2 * error;
85
86 if (double_error > -delta_y) {
87 error -= delta_y;
88 x1 += step_x;
89 }
90
91 if (double_error < delta_x) {
92 error += delta_x;
93 y1 += step_y;
94 }
95 }
96 }
