190 template<typename T>
191 void Image<T>::getROI(Image<T> &roiImg, const Rectangle &roiRect) {
192 if (roiRect.getX() + roiRect.getWidth() > width || roiRect.getY() + roiRect.getHeight() > height) {
193 throw std::invalid_argument("Rectangle is not within the bounds of the image!");
194 }
195 roiImg.zeroes(roiRect.getWidth(), roiRect.getHeight());
196 for (int i = 0; i < roiRect.getWidth(); ++i) {
197 for (int j = 0; j < roiRect.getHeight(); ++j) {
198 roiImg.at(i, j) = this->at(roiRect.getX() + i, roiRect.getY() + j);
199 }
200 }
